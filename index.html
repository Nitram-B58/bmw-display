// ── BMW M LOGO FIXED ───────────────────────────────────── function drawMLogoReal(c,s){ const cx=s/2,cy=s/2; c.clearRect(0,0,s,s); c.fillStyle='#000';c.fillRect(0,0,s,s); // Outer ring c.beginPath();c.arc(cx,cy,s*.46,0,Math.PI*2); const rg=c.createLinearGradient(cx-s*.46,cy,cx+s*.46,cy); rg.addColorStop(0,'#1a1a1a');rg.addColorStop(.5,'#444');rg.addColorStop(1,'#1a1a1a'); c.strokeStyle=rg;c.lineWidth=s*.022;c.stroke(); // The real BMW M logo layout: // Total width: ~72% of screen // Left portion: 3 italic stripes (~38% of total) // Right portion: Bold M letter (~62% of total) // Height: ~30% of screen, centered vertically const totalW = s * 0.72; const totalH = s * 0.30; const startX = cx - totalW/2; const startY = cy - totalH/2 - s*0.01; const slant = totalH * 0.5; // italic slant offset const stripeSection = totalW * 0.36; const sw = stripeSection / 3; // Stripe 1 - lighter blue #4f8ec9 c.beginPath(); c.moveTo(startX + slant, startY); c.lineTo(startX + slant + sw, startY); c.lineTo(startX + sw, startY + totalH); c.lineTo(startX, startY + totalH); c.closePath(); c.fillStyle = '#4a8ec8'; c.fill(); // Stripe 2 - dark navy #1d3f7a c.beginPath(); c.moveTo(startX + slant + sw, startY); c.lineTo(startX + slant + sw*2, startY); c.lineTo(startX + sw*2, startY + totalH); c.lineTo(startX + sw, startY + totalH); c.closePath(); c.fillStyle = '#1c3c78'; c.fill(); // Stripe 3 - red #c8171e c.beginPath(); c.moveTo(startX + slant + sw*2, startY); c.lineTo(startX + slant + sw*3, startY); c.lineTo(startX + sw*3, startY + totalH); c.lineTo(startX + sw*2, startY + totalH); c.closePath(); c.fillStyle = '#c8171e'; c.fill(); // M LETTER // Starts right after stripes with slight overlap const mStartX = startX + sw*2.6; const mEndX = startX + totalW; const mW = mEndX - mStartX; const mH = totalH; const mY = startY; const stemW = mW * 0.21; // stem/leg thickness c.fillStyle = '#ffffff'; c.shadowColor = 'rgba(255,255,255,0.15)'; c.shadowBlur = 6; // Left stem (vertical) c.fillRect(mStartX, mY, stemW, mH); // Right stem (vertical) c.fillRect(mEndX - stemW, mY, stemW, mH); // Left diagonal (top-left to center-bottom of V) // Goes from top-left inner edge to center at ~52% height c.beginPath(); c.moveTo(mStartX + stemW, mY); // top right of left stem c.lineTo(mStartX + stemW + stemW*0.3, mY); // slight width at top c.lineTo(mStartX + mW/2 + stemW*0.35, mY + mH*0.54); // center bottom right c.lineTo(mStartX + mW/2 - stemW*0.35, mY + mH*0.54); // center bottom left c.closePath(); c.fill(); // Right diagonal (top-right to center-bottom of V) c.beginPath(); c.moveTo(mEndX - stemW, mY); // top left of right stem c.lineTo(mEndX - stemW - stemW*0.3, mY); // slight width at top c.lineTo(mStartX + mW/2 - stemW*0.35, mY + mH*0.54); // center bottom left c.lineTo(mStartX + mW/2 + stemW*0.35, mY + mH*0.54); // center bottom right c.closePath(); c.fill(); c.shadowBlur = 0; } // ── NEON M FIXED ──────────────────────────────────────────
